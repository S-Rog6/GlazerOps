create table if not exists public.jobs (
  id bigint generated by default as identity primary key,

  po_number bigint null,
  job_name text not null,

  site_id bigint not null references public.sites(id),

  status text not null default 'Scheduled',
  scheduled_date date null,

  pinned boolean not null default false
);

create index if not exists idx_jobs_site_id on public.jobs(site_id);
create index if not exists idx_jobs_scheduled_date on public.jobs(scheduled_date);
create index if not exists idx_jobs_pinned on public.jobs(pinned);
