create table if not exists public.job_contacts (
  id bigint generated by default as identity primary key,
  job_id bigint not null references public.jobs(id) on delete cascade,

  name text not null,
  phone text null,
  email text null,
  role text null,     -- Super, PM, Foreman, Access, etc.
  notes text null,

  is_current boolean not null default true
);

create index if not exists idx_job_contacts_job_id
  on public.job_contacts(job_id);

create index if not exists idx_job_contacts_current
  on public.job_contacts(job_id) where is_current = true;

-- recommended: only one current per role per job
create unique index if not exists ux_job_contacts_current_role
on public.job_contacts(job_id, coalesce(role,''))
where is_current = true;
